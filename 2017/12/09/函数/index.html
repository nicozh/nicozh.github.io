函数（function）就是一组允许在你的代码里随时调用的语句，每个函数实际上就是一个短小的脚本。

函数声明

  1. 具名函数
function f(x,y){
     return x+y
 }
 f.name // 'f'
  1. 匿名函数
var f
 f = function(x,y){
     return x+y
 }
 f.name // 'f'
  1. 具名函数赋值
var f
 f = function f2(x,y){ return x+y }
 f.name // 'f2'
 console.log(f2) // undefined
  1. window.Function
var f = new Function('x','y','return x+y')
 f.name // "anonymous"
  1. 箭头函数
var f = (x,y) => {
     return x+y
 }
 var sum = (x,y) => x+y
 var n2 = n => n*n

变量作用域

  1. 全局变量（local variable）：可以在脚本中的任何位置被引用，包括函数内部。全局变量的作用域是整个脚本。
var a = 6;

function f(){
  console.log(a);
}

f()         // 6
  1. 局部变量（local variable）：只存在声明它的那个函数内部，在那个函数的外部无法引用它。局部变量的作用域仅限于某个特定的函数。
function f(){
  var a=1
}

a      // a is not defined
  1. 局部变量，会在该作用域内覆盖同名全局变量。需要注意的是，对于var命令来说，局部变量只能在函数内部声明，在其他区块中声明，一律都是全局变量。
var a = 6

function f(){
  var a = 7
  console.log(v)
}

f()       // 7
v        // 6
  1. 如果在某个函数中使用了var，那个变量将被视为一个局部变量,反之会被视为全局变量。如果脚本里已经存在一个与之同名的全局变量，这个函数会改变那个全局变量的值。
function f(x){
y=x*x                //不使用var，视为全局变量
return y
}

var y=60
var number=f(20)

console.log(y)               //  400

上面的y=x * x没有用var声明，它被当做赋值语句。它会寻找父作用域里有没有声明y，如果父作用域里声明了y，此处就是赋值；没有声明y就是声明一个全局变量y并赋值。

调用函数

f.call(asThis, input1,input2)
其中 asThis 会被当做 this，[input1,input2] 会被当做 arguments
不建议使用f(input1, input2)

this 和 arguments

function f(){
    'use strict'
    console.log(this)
    console.log(arguments)
    return undefined
}
f.call(1,2,3) // this 为 1，arguments 为 [2,3]

//末尾分号不能省略
(function(){ /* code */ }());
// 或者
(function(){ /* code */ })();
